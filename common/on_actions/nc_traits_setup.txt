on_game_started = {
	effect = {
		# Permanent unlock so the capitol can be (re)built even after the JE ends
		c:USA ?= {
			set_variable = nc_capitol_unlocked
		}

		# State traits live in map_data/state_regions/nc_flavor.txt (not add_state_trait here).
		s:STATE_NORTH_CAROLINA = {
			# Queue the State Capitol as under construction at game start (1836).
			# ~40% already done historically; remaining cost is reduced on the building.
			# Historically completed in 1840; tracked by je_nc_state_capitol.
			random_scope_state = {
				limit = {
					owner = c:USA
				}
				# Country prestige from Mount Mitchell (state traits can't carry country_* mods)
				owner = {
					if = {
						limit = {
							NOT = { has_modifier = nc_mount_mitchell_prestige }
						}
						add_modifier = {
							name = nc_mount_mitchell_prestige
						}
					}
				}
				if = {
					limit = {
						NOT = { has_building = building_nc_state_capitol }
					}
					start_building_construction = building_nc_state_capitol
					add_modifier = {
						name = nc_capitol_under_construction
						days = long_modifier_time
					}
					owner = {
						add_modifier = {
							name = nc_capitol_under_construction_country
							days = long_modifier_time
						}
					}
				}
			}
		}
	}
}

on_monthly_pulse_country = {
	events = {
		nc_flavor_events.7	# Charlotte Mint — July 27, 1837
		nc_flavor_events.9	# Outer Banks Lighthouses — from 1840
		nc_flavor_events.22	# Wright Brothers — December 17, 1903
	}
}
