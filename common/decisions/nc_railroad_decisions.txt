########################################
# Decisions to charter NC railroad lines
# State funding only — full queue to JE target.
# Private funding is via charter events (option B).
########################################

nc_charter_wilmington_weldon = {
	is_shown = {
		has_journal_entry = je_nc_wilmington_weldon
		NOT = { has_variable = nc_wilmington_weldon_chartered }
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
		}
	}

	possible = {
		gold_reserves >= 150000
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
			is_incorporated = yes
		}
	}

	when_taken = {
		add_treasury = -150000
		set_variable = nc_wilmington_weldon_chartered
		remove_variable = nc_wilmington_weldon_private
		nc_save_nc_rail_state = yes
		scope:nc_state_scope = {
			nc_queue_railways_to_level = { LEVEL = 8 }
			add_modifier = {
				name = nc_railroad_under_construction
				years = 1
			}
		}
		custom_tooltip = nc_charter_wilmington_weldon_tt
	}

	ai_chance = {
		value = 40
	}
}
