########################################
# Outer Banks Lighthouse System
# Sequential unlocks via choice event (like universities).
# Cape Hatteras rebuild remains a separate upgrade decision.
########################################

nc_choose_next_lighthouse = {
	is_shown = {
		has_variable = nc_lighthouse_system_active
		NOT = { has_variable = nc_lighthouse_system_complete }
		NOT = { has_variable = nc_light_build_pending }
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
		}
	}

	possible = {
		has_variable = nc_lighthouse_system_active
		NOT = { has_variable = nc_light_build_pending }
		NOT = { has_variable = nc_lighthouse_system_complete }
		# At least one unlockable light's tech is researched and not yet founded
		OR = {
			AND = {
				has_technology_researched = lathe
				NOT = { has_variable = nc_light_founded_lookout }
			}
			AND = {
				has_technology_researched = railways
				NOT = { has_variable = nc_light_founded_bodie }
			}
			AND = {
				has_technology_researched = railways
				NOT = { has_variable = nc_light_founded_currituck }
			}
			AND = {
				has_technology_researched = electrical_generation
				NOT = { has_variable = nc_light_founded_cape_fear }
			}
		}
	}

	when_taken = {
		trigger_event = { id = nc_flavor_events.33 popup = yes }
	}

	ai_chance = {
		value = 20
	}
}

nc_start_lighthouse_program = {
	is_shown = {
		NOT = { has_variable = nc_lighthouse_system_active }
		NOT = { has_variable = nc_lighthouse_system_complete }
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
		}
	}

	possible = {
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
			is_incorporated = yes
		}
	}

	when_taken = {
		trigger_event = { id = nc_flavor_events.9 popup = yes }
	}

	ai_chance = {
		value = 10
	}
}

########################################
# Rebuild Cape Hatteras as the tall brick tower (~1870)
########################################

nc_rebuild_cape_hatteras = {
	is_shown = {
		NOT = { has_variable = nc_hatteras_rebuilt }
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
			has_building = building_nc_cape_hatteras_lighthouse
		}
		has_technology_researched = steel_railway_cars
	}

	possible = {
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
			is_incorporated = yes
			has_building = building_nc_cape_hatteras_lighthouse
		}
	}

	when_taken = {
		set_variable = nc_hatteras_rebuilt
		add_treasury = -45000
		activate_production_method = {
			building_type = building_nc_cape_hatteras_lighthouse
			production_method = pm_nc_cape_hatteras_rebuilt
		}
		custom_tooltip = nc_rebuild_cape_hatteras_tt
	}

	ai_chance = {
		value = 30
	}
}
