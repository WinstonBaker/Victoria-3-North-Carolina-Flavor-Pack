########################################
# Decisions to authorize later Outer Banks lighthouses
# Sets unlock variables so the unique monuments become buildable.
########################################

nc_authorize_cape_lookout_lighthouse = {
	is_shown = {
		NOT = { has_variable = nc_unlocked_cape_lookout }
		NOT = {
			any_scope_state = {
				state_region = s:STATE_NORTH_CAROLINA
				has_building = building_nc_cape_lookout_lighthouse
			}
		}
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
		}
		has_technology_researched = lathe
	}

	possible = {
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
			is_incorporated = yes
		}
	}

	when_taken = {
		set_variable = nc_unlocked_cape_lookout
		add_treasury = -25000
		custom_tooltip = nc_authorize_cape_lookout_tt
	}

	ai_chance = {
		value = 25
	}
}

########################################
# Rebuild Cape Hatteras as the tall brick tower (~1870)
########################################

nc_rebuild_cape_hatteras = {
	is_shown = {
		NOT = { has_variable = nc_hatteras_rebuilt }
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
			has_building = building_nc_cape_hatteras_lighthouse
		}
		has_technology_researched = steel_railway_cars
	}

	possible = {
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
			is_incorporated = yes
			has_building = building_nc_cape_hatteras_lighthouse
		}
	}

	when_taken = {
		set_variable = nc_hatteras_rebuilt
		add_treasury = -45000
		# Switch the monument to the tall brick-tower production method
		activate_production_method = {
			building_type = building_nc_cape_hatteras_lighthouse
			production_method = pm_nc_cape_hatteras_rebuilt
		}
		custom_tooltip = nc_rebuild_cape_hatteras_tt
	}

	ai_chance = {
		value = 30
	}
}
