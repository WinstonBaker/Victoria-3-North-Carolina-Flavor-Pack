nc_new_bern_soft_drink = {
	is_shown = {
		has_journal_entry = je_nc_carolina_brands
		NOT = { has_variable = nc_pepsi_sponsored }
		NOT = { has_company = company_type:company_pepsi_cola }
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
		}
	}

	possible = {
		gold_reserves >= 40000
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
			is_incorporated = yes
		}
	}

	when_taken = {
		add_treasury = -40000
		set_variable = nc_pepsi_sponsored
		custom_tooltip = nc_new_bern_soft_drink_tt
		s:STATE_NORTH_CAROLINA = {
			random_scope_state = {
				limit = { owner = ROOT }
				add_modifier = {
					name = nc_pepsi_new_bern_boom
					days = long_modifier_time
				}
				if = {
					limit = {
						NOT = { has_building = building_food_industry }
					}
					start_building_construction = building_food_industry
				}
			}
		}
		ig:ig_industrialists ?= {
			add_modifier = {
				name = nc_ig_patronage_minor
				days = normal_modifier_time
			}
		}
	}

	ai_chance = {
		value = 15
	}
}
