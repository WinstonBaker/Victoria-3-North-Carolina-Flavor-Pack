########################################
# University System decisions
# Charter slots after Research Triangle (capacity 1, upgradeable to 2)
########################################

nc_choose_next_university = {
	is_shown = {
		has_variable = nc_research_triangle_complete
		NOT = { has_variable = nc_complete_unc_system_done }
		has_variable = nc_uni_charter_capacity
		var:nc_uni_charter_count < var:nc_uni_charter_capacity
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
		}
	}

	possible = {
		has_variable = nc_research_triangle_complete
		NOT = { has_variable = nc_complete_unc_system_done }
		var:nc_uni_charter_count < var:nc_uni_charter_capacity
	}

	when_taken = {
		trigger_event = { id = nc_flavor_events.32 popup = yes }
	}

	ai_chance = {
		value = 25
	}
}

# Permanent upgrade: charter two campuses at once
nc_expand_university_board = {
	is_shown = {
		has_variable = nc_research_triangle_complete
		NOT = { has_variable = nc_complete_unc_system_done }
		has_variable = nc_uni_charter_capacity
		var:nc_uni_charter_capacity < 2
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
		}
	}

	possible = {
		has_variable = nc_research_triangle_complete
		var:nc_uni_charter_capacity < 2
		gold_reserves >= 100000
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
			is_incorporated = yes
		}
	}

	when_taken = {
		add_treasury = -100000
		add_modifier = {
			name = nc_university_board_authority_cost
			years = 5
		}
		set_variable = {
			name = nc_uni_charter_capacity
			value = 2
		}
		custom_tooltip = nc_expand_university_board_tt
	}

	ai_chance = {
		value = 12
	}
}

# Timed construction push while any campus charter is active
nc_university_construction_push = {
	is_shown = {
		has_variable = nc_research_triangle_complete
		NOT = { has_variable = nc_complete_unc_system_done }
		var:nc_uni_charter_count > 0
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
		}
	}

	possible = {
		var:nc_uni_charter_count > 0
		gold_reserves >= 50000
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
			is_incorporated = yes
			NOT = { has_modifier = nc_university_system_grants }
		}
	}

	when_taken = {
		add_treasury = -50000
		custom_tooltip = nc_university_construction_push_tt
		s:STATE_NORTH_CAROLINA = {
			random_scope_state = {
				limit = { owner = ROOT }
				add_modifier = {
					name = nc_university_system_grants
					days = normal_modifier_time
				}
			}
		}
	}

	ai_chance = {
		value = 15
	}
}


# Move Wake Forest College to Winston-Salem / university status
# Not at game start — needs railways + Piedmont growth (tobacco or mid-century date).
nc_relocate_wake_forest = {
	is_shown = {
		NOT = { has_variable = nc_wake_forest_relocated }
		has_technology_researched = railways
		OR = {
			game_date >= 1855.1.1
			has_variable = nc_research_triangle_complete
			any_scope_state = {
				state_region = s:STATE_NORTH_CAROLINA
				owner = ROOT
				any_scope_building = {
					is_building_type = building_tobacco_plantation
					level >= 3
				}
			}
		}
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
			has_building = building_wake_forest_college
		}
	}

	possible = {
		gold_reserves >= 40000
		has_technology_researched = railways
		NOT = { has_modifier = nc_wake_forest_offer_cooldown }
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
			is_incorporated = yes
			has_building = building_wake_forest_college
		}
	}

	when_taken = {
		trigger_event = { id = nc_flavor_events.39 popup = yes }
	}

	ai_chance = {
		value = 10
	}
}
