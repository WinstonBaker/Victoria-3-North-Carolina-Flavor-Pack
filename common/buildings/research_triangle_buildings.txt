# Level 1 = Main Campus; level 2 = Centennial (story-gated expand only).
# unique = no so level 2 is allowed; possible blocks casual expand / second copies.
building_nc_state_university = {
	building_group = bg_nc_flavor
	icon = "gfx/interface/icons/building_icons/ncsu.dds"
	expandable = yes
	buildable = yes
	downsizeable = no
	unique = no

	entity_not_constructed = { }
	entity_under_construction = { "building_construction_3x3_entity" }
	entity_constructed = { }

	city_gfx_interactions = {
		clear_size_area = yes
		size = 4
	}

	production_method_groups = {
		pmg_monument_nc_state
	}

	potential = {
		state_region = s:STATE_NORTH_CAROLINA
	}

	possible = {
		owner ?= {
			OR = {
				# Founding via Research Triangle (after delayed auto-queue event)
				AND = {
					has_variable = nc_research_triangle_unlocked
					NOT = {
						any_scope_state = {
							state_region = s:STATE_NORTH_CAROLINA
							has_building = building_nc_state_university
						}
					}
				}
				# Centennial expand: story-authorized, still level 1
				AND = {
					has_variable = nc_ncsu_centennial_authorized
					NOT = { has_variable = nc_ncsu_centennial_complete }
					any_scope_state = {
						state_region = s:STATE_NORTH_CAROLINA
						any_scope_building = {
							is_building_type = building_nc_state_university
							level = 1
						}
					}
				}
			}
		}
	}

	should_auto_expand = {
		always = no
	}

	required_construction = construction_cost_monument

	ai_value = 2000

	background = "gfx/interface/icons/building_icons/backgrounds/building_panel_bg_monuments.dds"
}
