# North Carolina State Capitol (1833–1840)
# Present at game start as under construction (~40% already done by 1836).
########################################

building_nc_state_capitol = {
	building_group = bg_nc_flavor
	icon = "gfx/interface/icons/building_icons/nc_state_capitol.dds"
	expandable = no
	buildable = yes
	downsizeable = no
	unique = yes

	entity_not_constructed = { }
	entity_under_construction = { "building_construction_3x3_entity" }
	entity_constructed = { }

	city_gfx_interactions = {
		clear_size_area = yes
		size = 3
	}

	production_method_groups = {
		pmg_monument_nc_capitol
	}

	potential = {
		state_region = s:STATE_NORTH_CAROLINA
	}

	# Unlocked at game start for USA; persists so the capitol can be rebuilt if lost
	possible = {
		owner ?= {
			OR = {
				has_variable = nc_capitol_unlocked
				has_journal_entry = je_nc_state_capitol
			}
		}
	}

	# Remaining work only — exterior walls largely stood by late 1834
	required_construction = construction_cost_nc_capitol_remaining

	ai_value = 5000

	background = "gfx/interface/icons/building_icons/backgrounds/building_panel_bg_monuments.dds"
}
