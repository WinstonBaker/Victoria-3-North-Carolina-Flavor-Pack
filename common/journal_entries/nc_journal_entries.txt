je_nc_tar_heel_legacy = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_nc_flavor

	is_shown_when_inactive = {
		NOT = { has_variable = nc_tar_heel_legacy_complete }
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			is_incorporated = yes
		}
	}

	is_shown_in_lobby = {
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
		}
	}

	possible = {
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			is_incorporated = yes
		}
	}

	immediate = {
		s:STATE_NORTH_CAROLINA = {
			random_scope_state = {
				limit = {
					owner = ROOT
				}
				save_scope_as = nc_state_scope
			}
		}
	}

	complete = {
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			is_incorporated = yes
			any_scope_building = {
				is_building_type = building_tobacco_plantation
				level >= 5
			}
			
			any_scope_building = {
				is_building_type = building_railway
				level >= 3
			}
		}
	}

	on_complete = {
		set_variable = nc_tar_heel_legacy_complete
		trigger_event = { id = nc_flavor_events.4 popup = yes }
	}

	invalid = {
		NOT = {
			any_scope_state = {
				state_region = s:STATE_NORTH_CAROLINA
				is_incorporated = yes
			}
		}
	}

	on_monthly_pulse = {
		random_events = {
			200 = 0
			10 = nc_flavor_events.1
			10 = nc_flavor_events.2
			10 = nc_flavor_events.3
		}
	}

	weight = 75

	should_be_pinned_by_default_uninvolved_or_context = yes
}

je_nc_research_triangle = {
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	group = je_group_nc_flavor

	is_shown_when_inactive = {
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
		}
	}

	possible = {
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
		}
	}

	immediate = {
		# Safely triggers your introductory event instantly without the broken days param
		trigger_event = { id = nc_flavor_events.5 popup = yes }
	}

	# Three separate condition blocks to force three independent checkmarks in the UI
	complete = {
		s:STATE_NORTH_CAROLINA = {
			has_building = building_nc_state_university
		}
		s:STATE_NORTH_CAROLINA = {
			has_building = building_duke_university
		}
		s:STATE_NORTH_CAROLINA = {
			has_building = building_unc_chapel_hill
		}
	}

	on_complete = {
		trigger_event = { id = nc_flavor_events.6 popup = yes } 
		add_modifier = {
			name = nc_research_triangle_country_bonus
		}
		every_scope_state = {
			limit = {
				state_region = s:STATE_NORTH_CAROLINA
			}
			add_modifier = {
				name = modifier_research_triangle_established
			}
		}
	}

	weight = 100
	should_be_pinned_by_default_uninvolved_or_context = yes
}
