			owner = ROOT

				desc = je_nc_railroad_ncr_status_loc
				trigger = {
					has_variable = nc_wilmington_weldon_done
					has_variable = nc_raleigh_gaston_done
				}
			}
			triggered_desc = {
				desc = je_nc_railroad_gaston_status_loc

					has_variable = nc_wilmington_weldon_done
				}
			}
			triggered_desc = {
				desc = je_nc_railroad_progress_status_loc

			text = je_nc_railroad_rails_tt

				state_region = s:STATE_NORTH_CAROLINA
				any_scope_building = {
					is_building_type = building_railway
					level >= 5

			}
		}
	}

########################################
# North Carolina State Capitol (finish by ~1840)
########################################

je_nc_state_capitol = {
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"
	group = je_group_nc_flavor

	is_shown_when_inactive = {
		NOT = { has_variable = nc_capitol_complete }
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
		}
	}

	possible = {
		NOT = { has_variable = nc_capitol_complete }
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
		}
	}

	immediate = {
		set_variable = nc_capitol_unlocked
		s:STATE_NORTH_CAROLINA = {
			random_scope_state = {
				limit = { owner = ROOT }
				save_scope_as = nc_state_scope
				# Construction is already queued at game start via on_game_started.
				# Only re-queue if somehow missing (e.g. non-USA owner mid-game).
				if = {
					limit = {
						NOR = {
							has_building = building_nc_state_capitol
							# Avoid double-queue when already building from 1836 start
							has_modifier = nc_capitol_under_construction
						}
					}
					start_building_construction = building_nc_state_capitol
					add_modifier = {
						name = nc_capitol_under_construction
						days = long_modifier_time
					}
					owner = {
						add_modifier = {
							name = nc_capitol_under_construction_country
							days = long_modifier_time
						}
					}
				}
			}
		}
	}

	complete = {
		custom_tooltip = {
			text = je_nc_state_capitol_tt
			any_scope_state = {
				state_region = s:STATE_NORTH_CAROLINA
				has_building = building_nc_state_capitol
			}
		}
	}

	on_complete = {
		set_variable = nc_capitol_complete
		# Clear construction modifiers as soon as the building exists (event also cleans up)
		every_scope_state = {
			limit = {
				state_region = s:STATE_NORTH_CAROLINA
				has_modifier = nc_capitol_under_construction
			}
			remove_modifier = nc_capitol_under_construction
		}
		if = {
			limit = { has_modifier = nc_capitol_under_construction_country }
			remove_modifier = nc_capitol_under_construction_country
		}
		trigger_event = { id = nc_flavor_events.8 popup = yes }
	}

	invalid = {
		NOT = {
			any_scope_state = {
				state_region = s:STATE_NORTH_CAROLINA
				owner = ROOT
			}
		}
	}

	weight = 90
	should_be_pinned_by_default_uninvolved_or_context = yes
}
