je_nc_tar_heel_legacy = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"
	group = je_group_nc_flavor

	is_shown_when_inactive = {
		NOT = { has_variable = nc_tar_heel_legacy_complete }
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			is_incorporated = yes
		}
	}

	is_shown_in_lobby = {
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
		}
	}

	possible = {
		NOT = { has_variable = nc_tar_heel_legacy_complete }
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			is_incorporated = yes
		}
	}

	immediate = {
		s:STATE_NORTH_CAROLINA = {
			random_scope_state = {
				limit = { owner = ROOT }
				save_scope_as = nc_state_scope
			}
		}
	}

	complete = {
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			is_incorporated = yes
			any_scope_building = {
				is_building_type = building_tobacco_plantation
				level >= 5
			}
			any_scope_building = {
				is_building_type = building_logging_camp
				level >= 3
			}
		}
	}

	on_complete = {
		set_variable = nc_tar_heel_legacy_complete
		trigger_event = { id = nc_flavor_events.4 popup = yes }
	}

	invalid = {
		NOT = {
			any_scope_state = {
				state_region = s:STATE_NORTH_CAROLINA
				is_incorporated = yes
			}
		}
	}

	on_monthly_pulse = {
		random_events = {
			200 = 0
			10 = nc_flavor_events.1
			10 = nc_flavor_events.2
		}
	}

	weight = 75
	should_be_pinned_by_default_uninvolved_or_context = yes
}

# Layer 1 — Wilmington & Weldon Railroad
# Charter via Decision, then build Railways to level 8
########################################

je_nc_wilmington_weldon = {
	icon = "gfx/interface/icons/event_icons/event_industry.dds"
	group = je_group_nc_flavor

	is_shown_when_inactive = {
		NOT = { has_variable = nc_wilmington_weldon_done }
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
		}
	}

	possible = {
		NOT = { has_variable = nc_wilmington_weldon_done }
		has_technology_researched = railways
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
		}
	}

	immediate = {
		nc_save_nc_rail_state = yes
	}

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_nc_wilmington_weldon_building_status_loc
				trigger = { has_variable = nc_wilmington_weldon_chartered }
			}
			triggered_desc = {
				desc = je_nc_wilmington_weldon_progress_status_loc
				trigger = { always = yes }
			}
		}
	}

	complete = {
		custom_tooltip = {
			text = je_nc_wilmington_weldon_charter_tt
			has_variable = nc_wilmington_weldon_chartered
		}
		custom_tooltip = {
			text = je_nc_wilmington_weldon_rails_tt
			any_scope_state = {
				state_region = s:STATE_NORTH_CAROLINA
				any_scope_building = {
					is_building_type = building_railway
					level >= 8
				}
			}
		}
	}

	on_complete = {
		set_variable = nc_wilmington_weldon_done
		every_scope_state = {
			limit = { state_region = s:STATE_NORTH_CAROLINA }
			if = {
				limit = { has_modifier = nc_railroad_under_construction }
				remove_modifier = nc_railroad_under_construction
			}
		}
		if = {
			limit = { has_variable = nc_wilmington_weldon_private }
			every_scope_state = {
				limit = { state_region = s:STATE_NORTH_CAROLINA }
				add_modifier = {
					name = nc_wilmington_weldon_minor
					years = 2
				}
			}
		}
		else = {
			every_scope_state = {
				limit = { state_region = s:STATE_NORTH_CAROLINA }
				add_modifier = {
					name = nc_wilmington_weldon
					years = 2
				}
			}
		}
	}

	# Optional charter via event if the Decision was not taken
	on_monthly_pulse = {
		random_events = {
			150 = 0
			25 = nc_flavor_events.10
		}
	}

	invalid = {
		NOT = {
			any_scope_state = {
				state_region = s:STATE_NORTH_CAROLINA
				owner = ROOT
			}
		}
	}

	weight = 100
	should_be_pinned_by_default_uninvolved_or_context = yes
}

# Research Triangle — shown in Potential early; unlocks after Tar Heel Legacy
########################################

je_nc_research_triangle = {
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	group = je_group_nc_flavor

	is_shown_when_inactive = {
		NOT = { has_variable = nc_research_triangle_complete }
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
		}
	}

	is_shown_in_lobby = {
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
		}
	}

	possible = {
		NOT = { has_variable = nc_research_triangle_complete }
		custom_tooltip = {
			text = je_nc_research_triangle_tar_heel_tt
			has_variable = nc_tar_heel_legacy_complete
		}
		any_scope_state = {
			state_region = s:STATE_NORTH_CAROLINA
			owner = ROOT
		}
	}

	immediate = {
		# Unlock buildability as soon as the JE is active (matches Biltmore pattern).
		# Funding event still grants construction bonuses; event 40 auto-queues later.
		set_variable = nc_research_triangle_unlocked
		s:STATE_NORTH_CAROLINA = {
			random_scope_state = {
				limit = { owner = ROOT }
				save_scope_as = nc_state_scope
			}
		}
		# One-shot opener; re-queue auto-build if funding was committed but auto-queue missed.
		if = {
			limit = {
				NOT = { has_variable = nc_research_triangle_proposal_fired }
			}
			trigger_event = { id = nc_flavor_events.5 popup = yes }
		}
		else_if = {
			limit = {
				has_variable = nc_research_triangle_funding_committed
				has_variable = nc_research_triangle_unlocked
				NOT = { has_variable = nc_research_triangle_autoqueued }
			}
			trigger_event = {
				id = nc_flavor_events.40
				months = 1
			}
		}
	}

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_nc_research_triangle_waiting_status_loc
				trigger = {
					has_variable = nc_research_triangle_funding_committed
					has_variable = nc_research_triangle_unlocked
					NOT = { has_variable = nc_research_triangle_autoqueued }
				}
			}
			triggered_desc = {
				desc = je_nc_research_triangle_complete_status_loc
				trigger = {
					any_scope_state = {
						state_region = s:STATE_NORTH_CAROLINA
						has_building = building_nc_state_university
					}
					any_scope_state = {
						state_region = s:STATE_NORTH_CAROLINA
						has_building = building_duke_university
					}
					any_scope_state = {
						state_region = s:STATE_NORTH_CAROLINA
						has_building = building_unc_chapel_hill
					}
				}
			}
			triggered_desc = {
				desc = je_nc_research_triangle_nc_state_only_status_loc
				trigger = {
					any_scope_state = {
						state_region = s:STATE_NORTH_CAROLINA
						has_building = building_nc_state_university
					}
					NOT = {
						any_scope_state = {
							state_region = s:STATE_NORTH_CAROLINA
							has_building = building_duke_university
						}
					}
				}
			}
			triggered_desc = {
				desc = je_nc_research_triangle_duke_only_status_loc
				trigger = {
					any_scope_state = {
						state_region = s:STATE_NORTH_CAROLINA
						has_building = building_duke_university
					}
					NOT = {
						any_scope_state = {
							state_region = s:STATE_NORTH_CAROLINA
							has_building = building_nc_state_university
						}
					}
				}

				desc = je_nc_research_triangle_progress_status_loc

				state_region = s:STATE_NORTH_CAROLINA
				has_building = building_unc_chapel_hill
			}
		}
		custom_tooltip = {
			text = je_nc_research_triangle_ncsu_tt
			any_scope_state = {
				state_region = s:STATE_NORTH_CAROLINA
				has_building = building_nc_state_university
			}
		}
		custom_tooltip = {
			text = je_nc_research_triangle_duke_tt
			any_scope_state = {
				state_region = s:STATE_NORTH_CAROLINA
				has_building = building_duke_university
			}
		}
	}

	}

	event_outcome_completed_desc = {
		first_valid = {

				desc = event_effects_nc_research_triangle.6_tt

	invalid = {
