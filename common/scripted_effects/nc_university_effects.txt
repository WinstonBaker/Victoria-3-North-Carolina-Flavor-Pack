# Charter campus registry (UNC System regionals only — 14 campuses)
# id | build_var | founded_var | building | special_founded_effect

nc_uni_clamp_charter_count = {
	if = {
		limit = {
			has_variable = nc_uni_charter_count
			var:nc_uni_charter_count < 0
		}
		set_variable = {
			name = nc_uni_charter_count
			value = 0
		}
	}
}

# After a campus founds: bump count and fire mid-tier once at 7/14
nc_on_university_campus_founded = {
	if = {
		limit = {
			NOT = { has_variable = nc_uni_founded_count }
		}
		set_variable = {
			name = nc_uni_founded_count
			value = 0
		}
	}
	change_variable = {
		name = nc_uni_founded_count
		add = 1
	}
	if = {
		limit = {
			NOT = { has_variable = nc_university_system_expanding_done }
			var:nc_uni_founded_count >= 7
		}
		set_variable = nc_university_system_expanding_done
		trigger_event = { id = nc_flavor_events.38 popup = yes }
	}
}

# Fire system-complete reward once (Found JE or Progress JE)
nc_try_complete_university_system = {
	if = {
		limit = {
			NOT = { has_variable = nc_complete_unc_system_done }
			has_variable = nc_uni_founded_app_state
			has_variable = nc_uni_founded_ecu
			has_variable = nc_uni_founded_ecsu
			has_variable = nc_uni_founded_fayetteville
			has_variable = nc_uni_founded_ncat
			has_variable = nc_uni_founded_nccu
			has_variable = nc_uni_founded_asheville
			has_variable = nc_uni_founded_charlotte
			has_variable = nc_uni_founded_uncg
			has_variable = nc_uni_founded_pembroke
			has_variable = nc_uni_founded_uncsa
			has_variable = nc_uni_founded_wilmington
			has_variable = nc_uni_founded_western
			has_variable = nc_uni_founded_wssu
		}
		set_variable = nc_complete_unc_system_done
		if = {
			limit = {
				NOT = { has_variable = nc_uni_system_reward_fired }
			}
			set_variable = nc_uni_system_reward_fired
			trigger_event = { id = nc_flavor_events.31 popup = yes }
		}
	}
}

# Monthly maintenance: recover cancels, credit finishes, maybe complete system
nc_university_system_monthly = {
	if = {
		limit = {
			NOT = { has_variable = nc_uni_charter_count }
		}
		set_variable = {
			name = nc_uni_charter_count
			value = 0
		}
	}
	if = {
		limit = {
			NOT = { has_variable = nc_uni_charter_capacity }
		}
		set_variable = {
			name = nc_uni_charter_capacity
			value = 1
		}
	}
	nc_recover_stale_university_charters = yes
	nc_credit_finished_university_charters = yes
	nc_try_complete_university_system = yes
}
