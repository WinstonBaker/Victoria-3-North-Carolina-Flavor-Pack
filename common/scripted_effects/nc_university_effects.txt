# After a campus founds: bump count and fire mid-tier once at 8/16
nc_on_university_campus_founded = {
	if = {
		limit = {
			NOT = { has_variable = nc_uni_founded_count }
		}
		set_variable = {
			name = nc_uni_founded_count
			value = 0
		}
	}
	change_variable = {
		name = nc_uni_founded_count
		add = 1
	}
	if = {
		limit = {
			NOT = { has_variable = nc_university_system_expanding_done }
			check_variable = {
				name = nc_uni_founded_count
				value >= 8
			}
		}
		set_variable = nc_university_system_expanding_done
		trigger_event = { id = nc_flavor_events.38 popup = yes }
	}
}

nc_apply_davidson_founded_modifiers = {
	add_modifier = {
		name = nc_davidson_founded_country
	}
	every_scope_state = {
		limit = { state_region = s:STATE_NORTH_CAROLINA }
		add_modifier = {
			name = nc_davidson_founded
		}
	}
}
