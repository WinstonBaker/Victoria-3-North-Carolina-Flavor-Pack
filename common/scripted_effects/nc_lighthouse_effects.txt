# Free the pending slot when its build flag is set but the light
# is neither finished nor under construction (cancel / failed queue).
nc_recover_stale_lighthouse_charters = {
	if = {
		limit = {
			has_variable = nc_light_build_lookout
			NOT = {
				any_scope_state = {
					state_region = s:STATE_NORTH_CAROLINA
					OR = {
						has_building = building_nc_cape_lookout_lighthouse
						any_scope_building = {
							is_building_type = building_nc_cape_lookout_lighthouse
							is_under_construction = yes
						}
					}
				}
			}
		}
		remove_variable = nc_light_build_lookout
		remove_variable = nc_light_build_pending
	}
	if = {
		limit = {
			has_variable = nc_light_build_bodie
			NOT = {
				any_scope_state = {
					state_region = s:STATE_NORTH_CAROLINA
					OR = {
						has_building = building_nc_bodie_island_lighthouse
						any_scope_building = {
							is_building_type = building_nc_bodie_island_lighthouse
							is_under_construction = yes
						}
					}
				}
			}
		}
		remove_variable = nc_light_build_bodie
		remove_variable = nc_light_build_pending
	}
	if = {
		limit = {
			has_variable = nc_light_build_currituck
			NOT = {
				any_scope_state = {
					state_region = s:STATE_NORTH_CAROLINA
					OR = {
						has_building = building_nc_currituck_beach_lighthouse
						any_scope_building = {
							is_building_type = building_nc_currituck_beach_lighthouse
							is_under_construction = yes
						}
					}
				}
			}
		}
		remove_variable = nc_light_build_currituck
		remove_variable = nc_light_build_pending
	}
	if = {
		limit = {
			has_variable = nc_light_build_cape_fear
			NOT = {
				any_scope_state = {
					state_region = s:STATE_NORTH_CAROLINA
					OR = {
						has_building = building_nc_cape_fear_lighthouse
						any_scope_building = {
							is_building_type = building_nc_cape_fear_lighthouse
							is_under_construction = yes
						}
					}
				}
			}
		}
		remove_variable = nc_light_build_cape_fear
		remove_variable = nc_light_build_pending
	}
}

# Credit finished lights when the build JE missed them (soft-lock fix).
# Mirrors university nc_credit_finished_university_charters: if the light
# stands, mark founded and clear pending so Choose Next is not blocked.
nc_credit_finished_lighthouse_charters = {
	if = {
		limit = {
			NOT = { has_variable = nc_light_founded_lookout }
			any_scope_state = {
				state_region = s:STATE_NORTH_CAROLINA
				has_building = building_nc_cape_lookout_lighthouse
			}
		}
		if = {
			limit = { has_variable = nc_light_build_lookout }
			remove_variable = nc_light_build_lookout
			remove_variable = nc_light_build_pending
			set_variable = nc_light_just_credited
		}
		set_variable = nc_light_founded_lookout
	}
	if = {
		limit = {
			NOT = { has_variable = nc_light_founded_bodie }
			any_scope_state = {
				state_region = s:STATE_NORTH_CAROLINA
				has_building = building_nc_bodie_island_lighthouse
			}
		}
		if = {
			limit = { has_variable = nc_light_build_bodie }
			remove_variable = nc_light_build_bodie
			remove_variable = nc_light_build_pending
			set_variable = nc_light_just_credited
		}
		set_variable = nc_light_founded_bodie
	}
	if = {
		limit = {
			NOT = { has_variable = nc_light_founded_currituck }
			any_scope_state = {
				state_region = s:STATE_NORTH_CAROLINA
				has_building = building_nc_currituck_beach_lighthouse
			}
		}
		if = {
			limit = { has_variable = nc_light_build_currituck }
			remove_variable = nc_light_build_currituck
			remove_variable = nc_light_build_pending
			set_variable = nc_light_just_credited
		}
		set_variable = nc_light_founded_currituck
	}
	if = {
		limit = {
			NOT = { has_variable = nc_light_founded_cape_fear }
			any_scope_state = {
				state_region = s:STATE_NORTH_CAROLINA
				has_building = building_nc_cape_fear_lighthouse
			}
		}
		if = {
			limit = { has_variable = nc_light_build_cape_fear }
			remove_variable = nc_light_build_cape_fear
			remove_variable = nc_light_build_pending
			set_variable = nc_light_just_credited
		}
		set_variable = nc_light_founded_cape_fear
	}

	# Same follow-up as je_nc_build_selected_lighthouse on_complete
	if = {
		limit = {
			has_variable = nc_light_just_credited
		}
		remove_variable = nc_light_just_credited
		if = {
			limit = {
				NOT = {
					AND = {
						has_variable = nc_light_founded_lookout
						has_variable = nc_light_founded_bodie
						has_variable = nc_light_founded_currituck
						has_variable = nc_light_founded_cape_fear
					}
				}
			}
			trigger_event = { id = nc_flavor_events.33 popup = yes }
		}
	}
}

# Monthly maintenance: recover cancels + credit finishes
nc_lighthouse_system_monthly = {
	nc_recover_stale_lighthouse_charters = yes
	nc_credit_finished_lighthouse_charters = yes
}
