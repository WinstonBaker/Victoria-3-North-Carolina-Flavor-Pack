# Scale country + NC state education prestige from live campus count.
# Unit modifiers are multiplied by nc_education_prestige_multiplier (capped).
nc_refresh_education_prestige = {
	if = {
		limit = { has_modifier = nc_education_prestige_country }
		remove_modifier = nc_education_prestige_country
	}
	every_scope_state = {
		limit = {
			state_region = s:STATE_NORTH_CAROLINA
			has_modifier = nc_education_prestige
		}
		remove_modifier = nc_education_prestige
	}
	if = {
		limit = {
			any_scope_state = {
				state_region = s:STATE_NORTH_CAROLINA
				owner = ROOT
			}
			nc_university_campus_count > 0
		}
		add_modifier = {
			name = nc_education_prestige_country
			multiplier = nc_education_prestige_multiplier
		}
		every_scope_state = {
			limit = {
				state_region = s:STATE_NORTH_CAROLINA
				owner = ROOT
			}
			add_modifier = {
				name = nc_education_prestige
				multiplier = nc_education_prestige_multiplier
			}
		}
		set_variable = {
			name = nc_university_campus_count_var
			value = nc_university_campus_count
		}
	}
	else_if = {
		limit = { has_variable = nc_university_campus_count_var }
		remove_variable = nc_university_campus_count_var
	}
}
